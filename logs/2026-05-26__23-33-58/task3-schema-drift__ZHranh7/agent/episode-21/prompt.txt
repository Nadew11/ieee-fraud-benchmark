import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.metrics import roc_auc_score
from sklearn.tree import DecisionTreeClassifier
from sklearn.ensemble import RandomForestClassifier, ExtraTreesClassifier
import xgboost as xgb

train_df = pd.read_csv('/data/train_identity.csv')
train_df.columns = [c.replace('-', '_') if '-' in c else c for c in train_df.columns]

target_col = 'isFraud'
feature_cols = [c for c in train_df.columns if c != target_col and c != 'TransactionID']

train_sub, val_sub = train_test_split(train_df, test_size=0.20, random_state=42, stratify=train_df[target_col])

num_cols = train_df[feature_cols].select_dtypes(exclude=['object']).columns.tolist()

X_train_num = train_sub[num_cols].fillna(0)
X_val_num = val_sub[num_cols].fillna(0)

print('Evaluating Decision Tree...')
dt = DecisionTreeClassifier(max_depth=4, random_state=42, class_weight='balanced')
dt.fit(X_train_num, train_sub[target_col])
auc_dt = roc_auc_score(val_sub[target_col], dt.predict_proba(X_val_num)[:, 1])
print(f'Decision Tree (numeric only, depth 4) AUC: {auc_dt:.6f}')

print('Evaluating Random Forest...')
rf = RandomForestClassifier(n_estimators=100, max_depth=6, random_state=42, class_weight='balanced')
rf.fit(X_train_num, train_sub[target_col])
auc_rf = roc_auc_score(val_sub[target_col], rf.predict_proba(X_val_num)[:, 1])
print(f'Random Forest (numeric only, depth 6) AUC: {auc_rf:.6f}')

print('Evaluating Extra Trees...')
et = ExtraTreesClassifier(n_estimators=100, max_depth=6, random_state=42, class_weight='balanced')
et.fit(X_train_num, train_sub[target_col])
auc_et = roc_auc_score(val_sub[target_col], et.predict_proba(X_val_num)[:, 1])
print(f'Extra Trees (numeric only, depth 6) AUC: {auc_et:.6f}')

print('Evaluating XGBoost...')
clf_xgb = xgb.XGBClassifier(n_estimators=100, max_depth=3, learning_rate=0.01, random_state=42, eval_metric='auc')
clf_xgb.fit(X_train_num, train_sub[target_col])
auc_xgb = roc_auc_score(val_sub[target_col], clf_xgb.predict_proba(X_val_num)[:, 1])
print(f'XGBoost (numeric only) AUC: {auc_xgb:.6f}')
