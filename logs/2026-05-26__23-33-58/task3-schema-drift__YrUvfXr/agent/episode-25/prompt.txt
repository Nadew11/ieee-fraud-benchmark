import pandas as pd
import numpy as np
from sklearn.model_selection import StratifiedKFold
from sklearn.metrics import roc_auc_score
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LogisticRegression
from sklearn.pipeline import make_pipeline
from sklearn.impute import SimpleImputer

train = pd.read_csv('/data/train_identity.csv')
y = train['isFraud']
X = train.drop(columns=['isFraud', 'TransactionID'])

# Let's map categorical to codes
cat_cols = [c for c in X.columns if X[c].dtype == 'O' or X[c].dtype == 'object']
for col in cat_cols:
    X[col] = X[col].astype('category').cat.codes

def eval_features(features, C_val):
    skf = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
    val_aucs = []
    for tr_idx, val_idx in skf.split(X, y):
        X_tr, X_val = X.iloc[tr_idx][features], X.iloc[val_idx][features]
        y_tr, y_val = y.iloc[tr_idx], y.iloc[val_idx]

        pipe = make_pipeline(SimpleImputer(strategy='median'), StandardScaler(), LogisticRegression(C=C_val, random_state=42))
        pipe.fit(X_tr, y_tr)
        pred = pipe.predict_proba(X_val)[:, 1]
        val_aucs.append(roc_auc_score(y_val, pred))
    return np.mean(val_aucs), np.std(val_aucs)

candidates = [
    ['id_05'],
    ['id_37'],
    ['id_05', 'id_37'],
    ['id_05', 'id_37', 'id_02'],
    ['id_05', 'id_37', 'id_20'],
    ['id_05', 'id_37', 'DeviceType'],
    ['id_05', 'id_37', 'id_32'],
    ['id_05', 'id_37', 'id_16'],
    ['id_05', 'id_37', 'id_32', 'id_16'],
    ['id_05', 'id_37', 'id_32', 'id_16', 'DeviceType'],
]

for c in [0.001, 0.01, 0.1, 1.0, 10.0]:
    print(f"--- C = {c} ---")
    for cand in candidates:
        mean_auc, std_auc = eval_features(cand, c)
        print(f"Features {cand}: CV AUC = {mean_auc:.4f} +/- {std_auc:.4f}")
