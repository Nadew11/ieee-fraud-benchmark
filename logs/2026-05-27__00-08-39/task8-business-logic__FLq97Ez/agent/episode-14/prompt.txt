import pandas as pd
import numpy as np

df = pd.read_csv('/data/transactions.csv')
fx = pd.read_csv('/data/fx_rates.csv')
fees = pd.read_csv('/data/fee_schedule.csv')

# Convert dates
df['transaction_date'] = pd.to_datetime(df['transaction_date'])
fx['date'] = pd.to_datetime(fx['date'])

# Build FX rates lookup
all_dates = pd.date_range(start=min(df['transaction_date'].min(), fx['date'].min()),
                          end=max(df['transaction_date'].max(), fx['date'].max()))

fx_lookup = {'USD': pd.Series(1.0, index=all_dates)}
for cur in fx['currency'].unique():
    cur_rates = fx[fx['currency'] == cur].set_index('date')['rate_to_usd'].sort_index()
    fx_lookup[cur] = cur_rates.reindex(all_dates).ffill()

df['fx_rate'] = df.apply(lambda r: fx_lookup[r['currency']].loc[r['transaction_date']], axis=1)
df['amount_usd'] = df['TransactionAmt'] * df['fx_rate']

fee_dict = fees.set_index('merchant_category')['fee_rate'].to_dict()
df['fee_rate'] = df['merchant_category'].map(fee_dict)
df['processing_fee_usd'] = df['amount_usd'] * df['fee_rate']

df_q4 = df[(df['transaction_date'] >= '2023-10-01') & (df['transaction_date'] <= '2023-12-31')]

# Let's select all isFraud == 1 and purchase
purchases = df_q4[(df_q4['isFraud'] == 1) & (df_q4['transaction_type'] == 'purchase')]

print("Total purchase fraud in Q4:", len(purchases))

# Let's check differences of single transactions or subsets
# Target gross is 30212.56
target_gross = 30212.56

# Let's try subtracting each transaction or combinations
# Let's see if any subset of settled purchases sums close to target_gross
settled_purchases = purchases[purchases['settlement_status'] == 'settled'].copy()
print("Settled purchase count:", len(settled_purchases), "sum:", settled_purchases['amount_usd'].sum())

diff = settled_purchases['amount_usd'].sum() - target_gross
print("Difference from target_gross with all settled:", diff)

# Find if any single settled transaction has amount close to diff
close_to_diff = settled_purchases[(settled_purchases['amount_usd'] - diff).abs() < 10.0]
print("Transactions close to difference:")
print(close_to_diff[['TransactionID', 'transaction_date', 'amount_usd', 'settlement_status', 'currency']])
